1.	Выведите список студентов физико-математического факультета.

SELECT s.FIO
FROM Student s
WHERE s."Group" IN
    (SELECT g."Group"
    FROM G g
    WHERE g.Kafedra IN 
        (SELECT k.Kafedra
        FROM Kafedra k
        WHERE k.Decanat LIKE 'Физико-математический'
        )
    )

2.	Студенты какой кафедры и факультета сдавали социологию.

SELECT k.Decanat, k.Kafedra
FROM Kafedra k
WHERE k.Kafedra IN
    (SELECT g.Kafedra
    FROM G g
    WHERE g."Group" IN 
        (SELECT s."Group"
        FROM Student s
        WHERE s.ID_St IN
            (SELECT e.id_St
            FROM Exzamen e
            WHERE e.Predmet LIKE 'Социология'
            )
            ))

3.	Выведите кафедры, список студентов, список студентов в алфавитном порядке.

SELECT s.FIO,
    (SELECT g.Kafedra
    FROM G g
    WHERE s."Group"=g."Group") AS [Кафедра]
FROM Student s
ORDER BY s.FIO

14.	Подсчитать количество пятерок на физико-математическом факультете.

SELECT COUNT(e.Ball)  AS 'Кол-во 5-ок на факультете'
FROM Exzamen e
WHERE e.id_St IN
    (SELECT s.ID_St
    FROM Student s
    WHERE s."Group" IN
        (SELECT g."Group"
        FROM G g
        WHERE g.Kafedra IN
            (SELECT k.Kafedra
            FROM Kafedra k
            WHERE e.Ball =5 AND k.Decanat LIKE 'Физико-математический'
            )
        )
    )
